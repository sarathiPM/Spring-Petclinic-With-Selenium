# in pom
<plugin>
    <groupId>org.apache.jmeter</groupId>
    <artifactId>maven-jmeter-plugin</artifactId>
    <version>1.0</version>
    <executions>
        <execution>
            <id>jmeter-tests</id>
            <phase>verify</phase>
            <goals>
                <goal>jmeter</goal>
            </goals>
       </execution>
    </executions>
    <configuration>
        <reportDir>${project.build.directory}/jmeter-reports</reportDir>
        <jmeterUserProperties>
            <!-- for user properites -->
        </jmeterUserProperties>
    </configuration>
</plugin>
